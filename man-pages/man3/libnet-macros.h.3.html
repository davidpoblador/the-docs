#define LIBNET_DONT_RESOLVE 0 Used for libnet's name resolution functions, specifies that no DNS lookups should be performed and the IP address should be kept in numeric form.

#define LIBNET_ERRBUF_SIZE 0x100 The libnet error buffer is 256 bytes long.

#define LIBNET_MAX_PACKET 0xffff The biggest an IP packet can be -- 65,535 bytes.

#define LIBNET_MAXOPTION_SIZE 0x28 IP and TCP options can be up to 40 bytes long.

#define LIBNET_OFF 1 Used several places, to specify 'on' or 'one'

#define LIBNET_ON 0 Used several places, to specify 'on' or 'one'

#define LIBNET_PR2 0 Used for \fBlibnet_get_prand()\fP to specify function disposition

#define LIBNET_RESOLVE 1 Used for libnet's name resolution functions, specifies that a DNS lookup can be performed if needed to resolve the IP address to a canonical form.
