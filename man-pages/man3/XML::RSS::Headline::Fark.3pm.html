SYNOPSIS

Strip out the extra Fark redirect \s-1URL\s0 and strip out the various [blahblah] blocks in the headline

    use XML::RSS::Feed;
    use XML::RSS::Headline::Fark;
    use LWP::Simple qw(get);

my $feed = XML::RSS::Feed->new( name => "fark", url => "http://www.pluck.com/rss/fark.rss", hlobj => "XML::RSS::Headline::Fark", );

while (1) { $feed->parse(get($feed->url)); print $_->headline . "\n" for $feed->late_breaking_news; sleep($feed->delay); }
